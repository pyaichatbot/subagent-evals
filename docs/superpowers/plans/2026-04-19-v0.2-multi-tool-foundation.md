# v0.2: Multi-Tool Foundation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make `subagent-evals` work across all major markdown-agent runtimes — Claude Code, Codex, GitHub Copilot, Cursor, and Windsurf — with zero-config auto-detection and a publishable npm package.

**Architecture:** New per-format adapter files in `packages/core/src/adapters/` export a single `discover*Agents` function each. The central `discoverAgents` dispatcher in `packages/core/src/index.ts` is extended to call each adapter. A new `auto.ts` adapter runs all adapters in parallel and applies the stable-ID + dedup policy from the spec. A new `DiscoveryFormat` type (`AgentFormatId | "auto"`) replaces the `AgentFormatId` in the config's `format` field.

**Tech Stack:** TypeScript 6, Node.js 22+, ESM (`"type": "module"`), `fast-glob`, `js-yaml`, `vitest` — all already installed.

  ```typescript
  import { existsSync } from "node:fs";
  import { readFile } from "node:fs/promises";
  import { resolve } from "node:path";

  import fg from "fast-glob";

  import type { DiscoveredAgent } from "../index.js";

  export async function discoverCodexAgents(input: {
    cwd: string;
    roots: string[];
  }): Promise<DiscoveredAgent[]> {
    const results: DiscoveredAgent[] = [];

    for (const root of input.roots) {
      const absoluteRoot = resolve(input.cwd, root);

      // File-based: .codex/agents/*.md
      const agentsDir = resolve(absoluteRoot, ".codex/agents");
      if (existsSync(agentsDir)) {
        const paths = await fg("**/*.md", { cwd: agentsDir, absolute: true });
        results.push(
          ...paths.map((path) => ({
            path,
            format_adapter_id: "codex-md" as const
          }))
        );
      }

      // Section-based: AGENTS.md
      const agentsMd = resolve(absoluteRoot, "AGENTS.md");
      if (existsSync(agentsMd)) {
        const body = await readFile(agentsMd, "utf8");
        const sections = extractSections(body);
        results.push(
          ...sections.map(({ heading, content }) => ({
            path: agentsMd,
            format_adapter_id: "codex-md" as const,
            body: content,
            section: heading
          }))
        );
      }
    }

    return results;
  }

  interface Section {
    heading: string;
    content: string;
  }

  function extractSections(text: string): Section[] {
    const lines = text.split("\n");
    const sections: Section[] = [];
    let currentHeading: string | null = null;
    const currentLines: string[] = [];

    for (const line of lines) {
      const match = line.match(/^##\s+Agent:\s+(.+)$/i);
      if (match) {
        if (currentHeading !== null) {
          sections.push({ heading: currentHeading, content: currentLines.join("\n").trim() });
          currentLines.length = 0;
        }
        currentHeading = match[1].trim();
      } else if (currentHeading !== null) {
        currentLines.push(line);
      }
    }

    if (currentHeading !== null) {
      sections.push({ heading: currentHeading, content: currentLines.join("\n").trim() });
    }

    return sections;
  }
  ```

  ```typescript
  import { existsSync } from "node:fs";
  import { readFile } from "node:fs/promises";
  import { resolve } from "node:path";

  import type { DiscoveredAgent } from "../index.js";

  export async function discoverCopilotAgents(input: {
    cwd: string;
    roots: string[];
  }): Promise<DiscoveredAgent[]> {
    const results: DiscoveredAgent[] = [];

    for (const root of input.roots) {
      const filePath = resolve(input.cwd, root, ".github/copilot-instructions.md");
      if (!existsSync(filePath)) continue;

      const body = await readFile(filePath, "utf8");
      const sections = extractAgentSections(body);

      if (sections.length > 0) {
        results.push(
          ...sections.map(({ heading, content }) => ({
            path: filePath,
            format_adapter_id: "copilot-instructions" as const,
            body: content,
            section: heading
          }))
        );
      } else {
        // Whole file is one agent
        results.push({
          path: filePath,
          format_adapter_id: "copilot-instructions" as const,
          body
        });
      }
    }

    return results;
  }

  function extractAgentSections(text: string): Array<{ heading: string; content: string }> {
    const lines = text.split("\n");
    const sections: Array<{ heading: string; content: string }> = [];
    let currentHeading: string | null = null;
    const currentLines: string[] = [];

    for (const line of lines) {
      const match = line.match(/^##\s+Agent:\s+(.+)$/i);
      if (match) {
        if (currentHeading !== null) {
          sections.push({ heading: currentHeading, content: currentLines.join("\n").trim() });
          currentLines.length = 0;
        }
        currentHeading = match[1].trim();
      } else if (currentHeading !== null) {
        currentLines.push(line);
      }
    }

    if (currentHeading !== null) {
      sections.push({ heading: currentHeading, content: currentLines.join("\n").trim() });
    }

    return sections;
  }
  ```

  ```typescript
  import { existsSync } from "node:fs";
  import { resolve } from "node:path";

  import fg from "fast-glob";

  import type { DiscoveredAgent } from "../index.js";

  export async function discoverWindsurfAgents(input: {
    cwd: string;
    roots: string[];
  }): Promise<DiscoveredAgent[]> {
    const results: DiscoveredAgent[] = [];

    for (const root of input.roots) {
      const rulesDir = resolve(input.cwd, root, ".windsurf/rules");
      if (!existsSync(rulesDir)) continue;

      const paths = await fg("**/*.md", { cwd: rulesDir, absolute: true });
      results.push(
        ...paths.map((path) => ({
          path,
          format_adapter_id: "windsurf-config" as const
        }))
      );
    }

    return results;
  }
  ```

  ```typescript
  import { relative } from "node:path";

  import { discoverCodexAgents } from "./codex-md.js";
  import { discoverCopilotAgents } from "./copilot-instructions.js";
  import { discoverCursorAgents } from "./cursor-rules.js";
  import { discoverWindsurfAgents } from "./windsurf-config.js";
  import type { DiscoveredAgent } from "../index.js";

  // Claude adapter is in index.ts — import the internal function via a re-export trick.
  // We call discoverAgents with format:"claude-md" to avoid circular imports.
  import { discoverAgents } from "../index.js";

  export async function discoverAutoAgents(input: {
    cwd: string;
    roots: string[];
    globs?: string[] | undefined;
  }): Promise<DiscoveredAgent[]> {
    const [claudeAgents, codexAgents, copilotAgents, cursorAgents, windsurfAgents] =
      await Promise.all([
        discoverAgents({ ...input, format: "claude-md" }),
        discoverCodexAgents(input),
        discoverCopilotAgents(input),
        discoverCursorAgents(input),
        discoverWindsurfAgents(input)
      ]);

    const allAgents = [
      ...claudeAgents,
      ...codexAgents,
      ...copilotAgents,
      ...cursorAgents,
      ...windsurfAgents
    ];

    const activeAdapters = new Set(allAgents.map((a) => a.format_adapter_id));
    if (activeAdapters.size >= 3) {
      process.stderr.write(
        `[subagent-evals] Warning: 3 or more agent formats detected ` +
          `(${[...activeAdapters].join(", ")}). ` +
          `Add discovery.dedup: true and discovery.primary to resolve duplicates.\n`
      );
    }

    return allAgents.map((agent) => ({
      ...agent,
      id_override: buildQualifiedId(agent, input.cwd)
    }));
  }

  function buildQualifiedId(agent: DiscoveredAgent, cwd: string): string {
    const rel = relative(cwd, agent.path).replace(/\\/g, "/");
    const base = `${agent.format_adapter_id}:${rel}`;
    return agent.section ? `${base}#${agent.section}` : base;
  }
  ```

  > **Note on circular imports**: `auto.ts` imports `discoverAgents` from `index.ts`, and `index.ts` imports `discoverAutoAgents` from `auto.ts`. Node.js ESM handles this via lazy binding as long as neither module calls the imported function at module load time (only inside async functions). This is safe here because both imports are used only in async function bodies.

  ````markdown
  ## What it does

  | Cursor | `.cursor/rules/*.mdc` | `cursor-rules` |
  | Windsurf | `.windsurf/rules/*.md` | `windsurf-config` |
  | Generic | Any YAML-frontmatter `.md` glob | `generic-frontmatter-md` |

  ## Quickstart

  ```bash
  # Zero-config: auto-detects all formats in your repo
  npx subagent-evals@latest lint .

  # Initialize config
  npx subagent-evals@latest init

  # Full eval with runtime assertions

  # Generate HTML report
  ````
